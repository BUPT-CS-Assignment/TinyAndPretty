SQL Form of NEDB



Data Type Support >

[ int ] >>> int
[ int64 ] >>> long int
[ real ] >>> double
[ text ] >>> char[32]
[ longtext ] >>> char[255]


SQL FORM >

[TABLE CREATE]

create table 'table_name' ('data_title' 'data_type', ... );

* The first parm will be set as PRIMARY KEY by default.
* Add 'key' after a parm to designate.
* Parm typed 'longtext' is not allowed to be set as the PRIMARY KEY.


[TABLE REMOVE]

drop table 'table_name';


[DATA INSERT]

insert into 'table_name' ('parm_name', ...) values ('parm_value', ...);


[DATA SELECT]

select 'parm_name' from 'table_name' where 'conditions';


[DATA DELETE]

delete from 'table_name' where 'conditions';


[DATA UPDATE]

update 'table_name' set 'parm_name' = 'parm_value', ... where 'conditions';


[STRUCTURE CHECK]

describe table 'table_name';


[TABLES CHECK]

select tables;
